# 0. Add C# type for DWM API to enable Dark Mode on the window title bar ("handle")
if (-not ('DwmApi' -as [type])) {
    Add-Type -TypeDefinition @"
    using System;
    using System.Runtime.InteropServices;
    public class DwmApi {
        [DllImport("dwmapi.dll", PreserveSig = true)]
        public static extern int DwmSetWindowAttribute(IntPtr hwnd, int attr, ref int attrValue, int attrSize);
    }
"@
}

# 1. Define Static Dark Theme Color Palette
$WindowBg    = "#1E1E1E" # Very dark gray for the main window background
$ControlBg   = "#2D2D2D" # Lighter gray for control backgrounds (textboxes, buttons)
$TextFg      = "#F1F1F1" # Off-white for all text
$Border      = "#3F3F46" # Mid-gray for borders
$Accent      = "#007ACC" # Bright blue for primary actions and highlights
$AccentHover = "#0097FF" # Lighter blue for hover effects on accent elements
$DropBg      = "#004C7A" # Darker blue for the drag-drop hover effect

# 2. XAML Window.Resources for Dark Theme
<Window.Resources>
    <!-- Base Button Style -->
    <Style TargetType="{x:Type Button}">
        <Setter Property="Background" Value="$ControlBg" />
        <Setter Property="Foreground" Value="$TextFg" />
        <Setter Property="BorderBrush" Value="$Border" />
        <Setter Property="Padding" Value="8" />
        <Setter Property="BorderThickness" Value="1" />
        <Setter Property="Template">
            <Setter.Value>
                <ControlTemplate TargetType="{x:Type Button}">
                    <Border Background="{TemplateBinding Background}" BorderBrush="{TemplateBinding BorderBrush}" BorderThickness="{TemplateBinding BorderThickness}" CornerRadius="3">
                        <ContentPresenter HorizontalAlignment="Center" VerticalAlignment="Center"/>
                    </Border>
                </ControlTemplate>
            </Setter.Value>
        </Setter>
        <Style.Triggers>
            <Trigger Property="IsMouseOver" Value="True">
                <Setter Property="Background" Value="$Border" />
            </Trigger>
        </Style.Triggers>
    </Style>

    <!-- Accent Button Style (for Convert) -->
    <Style x:Key="AccentButton" TargetType="{x:Type Button}" BasedOn="{StaticResource {x:Type Button}}">
        <Setter Property="Background" Value="$Accent" />
        <Setter Property="Foreground" Value="#FFFFFF" />
        <Setter Property="BorderBrush" Value="$Accent" />
        <Style.Triggers>
            <Trigger Property="IsMouseOver" Value="True">
                <Setter Property="Background" Value="$AccentHover" />
                <Setter Property="BorderBrush" Value="$AccentHover" />
            </Trigger>
        </Style.Triggers>
    </Style>

    <!-- ComboBox Style -->
    <Style TargetType="{x:Type ComboBox}">
        <Setter Property="Background" Value="$ControlBg" />
        <Setter Property="Foreground" Value="$TextFg" />
        <Setter Property="BorderBrush" Value="$Border" />
        <Setter Property="Padding" Value="4" />
        <Setter Property="Template">
            <Setter.Value>
                <ControlTemplate TargetType="{x:Type ComboBox}">
                    <Grid>
                        <ToggleButton
                            Name="ToggleButton"
                            Focusable="false"
                            IsChecked="{Binding Path=IsDropDownOpen, Mode=TwoWay, RelativeSource={RelativeSource TemplatedParent}}"
                            Background="$ControlBg"
                            BorderBrush="$Border"
                            BorderThickness="{TemplateBinding BorderThickness}"
                            Grid.Column="2">
                            <Path Name="Arrow" Fill="$TextFg" HorizontalAlignment="Center" VerticalAlignment="Center" Data="M 0 0 L 4 4 L 8 0" />
                        </ToggleButton>
                        <ContentPresenter Name="ContentSite"
                            IsHitTestVisible="False"
                            Content="{TemplateBinding SelectionBoxItem}"
                            ContentTemplate="{TemplateBinding SelectionBoxItemTemplate}"
                            Margin="5,0,23,0"
                            VerticalAlignment="Center"
                            HorizontalAlignment="Left" />
                        <Popup Name="Popup"
                            Placement="Bottom"
                            IsOpen="{TemplateBinding IsDropDownOpen}"
                            AllowsTransparency="True"
                            Focusable="False"
                            PopupAnimation="Slide">
                            <Grid Name="DropDown" SnapsToDevicePixels="True" MinWidth="{TemplateBinding ActualWidth}" MaxHeight="{TemplateBinding MaxDropDownHeight}">
                                <Border Name="DropDownBorder"
                                    Background="$ControlBg"
                                    BorderThickness="1"
                                    BorderBrush="$Border" />
                                <ScrollViewer Margin="4" SnapsToDevicePixels="True">
                                    <StackPanel IsItemsHost="True" />
                                </ScrollViewer>
                            </Grid>
                        </Popup>
                    </Grid>
                </ControlTemplate>
            </Setter.Value>
        </Setter>
    </Style>

    <!-- ComboBox Item Style (Crucial for dropdown list) -->
    <Style TargetType="{x:Type ComboBoxItem}">
        <Setter Property="Background" Value="$ControlBg" />
        <Setter Property="Foreground" Value="$TextFg" />
        <Style.Triggers>
            <Trigger Property="IsHighlighted" Value="True">
                <Setter Property="Background" Value="$Accent" />
                <Setter Property="Foreground" Value="#FFFFFF" />
            </Trigger>
            <Trigger Property="IsEnabled" Value="False">
                <Setter Property="Foreground" Value="Gray" />
            </Trigger>
        </Style.Triggers>
    </Style>

    <!-- TextBox Style -->
    <Style TargetType="{x:Type TextBox}">
        <Setter Property="Background" Value="$ControlBg" />
        <Setter Property="Foreground" Value="$TextFg" />
        <Setter Property="BorderBrush" Value="$Border" />
        <Setter Property="BorderThickness" Value="1" />
        <Setter Property="Padding" Value="2" />
        <Setter Property="CaretBrush" Value="$TextFg" />
    </Style>

    <!-- Separator Style -->
    <Style TargetType="{x:Type Separator}">
        <Setter Property="Background" Value="$Border" />
        <Setter Property="Margin" Value="0,0,0,10" />
    </Style>
</Window.Resources>

# 3. Apply Dark Mode to Title Bar ("Handle")
# (Ensure $Window contains your loaded XAML Window object before running this)
$hwnd = (New-Object System.Windows.Interop.WindowInteropHelper($Window)).EnsureHandle()
$trueVal = 1
[DwmApi]::DwmSetWindowAttribute($hwnd, 20, [ref]$trueVal, 4) | Out-Null # Windows 10 1903+ and Windows 11
[DwmApi]::DwmSetWindowAttribute($hwnd, 19, [ref]$trueVal, 4) | Out-Null # Windows 10 1809
